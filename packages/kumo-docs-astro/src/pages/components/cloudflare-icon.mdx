<ComponentSection>
  <Heading level={2}>Usage</Heading>

```tsx
import { CloudflareIcon } from "@cloudflare/kumo";

export default function Example() {
  return (
    <CloudflareIcon
      glyph="cloudflare-gateway-outline"
      title="Cloudflare Gateway"
    />
  );
}
```

<p>
  <code>CloudflareIcon</code> renders real shipped glyphs from Kumo’s generated
  Cloudflare icon catalog. Use a glyph name from <code>cloudflareIconNames</code>{" "}
  and size or recolor the icon with normal Kumo classes.
</p>
</ComponentSection>
